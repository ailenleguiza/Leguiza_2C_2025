113void Funcion(void* param)
114{
115 uint16_t guardado;
116 while (true)
117 {
118 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
119 AnalogInputReadSingle(CH1,&guardado);
120 UartSendString(UART_PC, (char*)UartItoa(guardado, 10));
121 UartSendString(UART_PC, "\r");
122 }
123}
